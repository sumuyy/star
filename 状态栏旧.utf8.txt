    function renderConsole(sd) {
      if (!state.lastData) { state.lastData = _.cloneDeep(sd); return; }
      const changes = [];
      function compare(o, n, p) {
        if (_.isPlainObject(o) && _.isPlainObject(n)) Object.keys(n).forEach(k => compare(o[k], n[k], p ? `${p}.${k}` : k));
        else if (!_.isEqual(o, n)) changes.push(`<div class="console-line"><span class="sys-tag">SYS</span>${p} = ${_.isObject(n)?JSON.stringify(n):n}</div>`);
      }
      compare(state.lastData, sd, 'stat_data'); state.lastData = _.cloneDeep(sd);
      if (changes.length > 0) { $('#data-console-log').append(changes.join('')); const el = document.getElementById('data-console-log'); if(el) el.scrollTop = el.scrollHeight; }
    }
